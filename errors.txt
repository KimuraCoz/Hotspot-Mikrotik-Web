# File pesan error untuk Yukari Hotspot
# Diterjemahkan ke Bahasa Indonesia

# internal-error
# Kesalahan internal yang jarang terjadi
internal-error = kesalahan internal ($(error-orig))

# config-error
# Kesalahan konfigurasi pada router
config-error = kesalahan konfigurasi ($(error-orig))

# not-logged-in
# Muncul jika user mengakses halaman status/logout tapi belum login
not-logged-in = Anda belum login (ip $(ip))

# ippool-empty
# Muncul jika IP Address di pool habis
ippool-empty = gagal memberikan alamat IP - pool alamat kosong

# shutting-down
# Muncul saat hotspot dimatikan
shutting-down = layanan hotspot sedang dimatikan

# user-session-limit
# Muncul jika user login melebihi batas "Shared Users"
user-session-limit = batas sesi tercapai, tidak bisa login lagi untuk user $(username)

# license-session-limit
# Batas lisensi routerOS tercapai
license-session-limit = batas lisensi sesi tercapai ($(error-orig))

# wrong-mac-username
# Jika username diset sebagai MAC address tapi login dari MAC berbeda
wrong-mac-username = username tidak valid ($(username)): MAC address ini bukan milik Anda

# chap-missing
# Biasanya karena JavaScript mati atau password tidak terkirim
chap-missing = browser tidak mengirim respon challenge (coba lagi, pastikan JavaScript aktif)

# invalid-username
# Pesan error paling umum (salah password/username)
# Kita buat lebih sopan dan jelas
invalid-username = Username atau password salah

# invalid-mac
# Jika user dikunci ke MAC address tertentu (Bind to MAC)
invalid-mac = user $(username) tidak diizinkan login dari perangkat ini

# uptime-limit, traffic-limit
# Jika batas waktu atau kuota user habis
uptime-limit = waktu akses user $(username) telah habis
traffic-limit = kuota trafik user $(username) telah habis

# radius-timeout
# Jika menggunakan server Radius dan tidak merespon
radius-timeout = server otentikasi tidak merespon

# auth-in-progress
# Jika user menekan login berkali-kali dengan cepat
auth-in-progress = otentikasi sedang berjalan, mohon tunggu...

# radius-reply
# Pesan custom dari Radius server
radius-reply = $(error-orig)
